    "i_serial_buffered_file_serializer.h"
    "i_serial_classid.h"
    "i_serial_file_serializer.h"
    "i_serial_serializable.h"

    "serial.h"
    "serial_inline.h"
    "serial_marker_helpers.h"
    )

set(PROJECT_SOURCES

    ${PROJECT_HEADERS}
