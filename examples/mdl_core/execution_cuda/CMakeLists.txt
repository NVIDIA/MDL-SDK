# name of the target and the resulting example
set(PROJECT_NAME examples-mdl_core-execution_cuda)

# -------------------------------------------------------------------------------------------------
# Main Target
# -------------------------------------------------------------------------------------------------

# collect sources
set(PROJECT_SOURCES
    "example_execution_cuda.cpp"
    )

if(MDL_BUILD_WITHOUT_CUDA_DRIVER)
    set(ADDITIONAL_OPTS ${ADDITIONAL_OPTS} SKIP_UNDEFINED_SYMBOL_CHECK)
endif()

# create target from template
create_from_base_preset(
    TARGET ${PROJECT_NAME}
    TYPE EXECUTABLE
    OUTPUT_NAME "execution_cuda"
    SOURCES ${PROJECT_SOURCES}
    EXAMPLE
    ${ADDITIONAL_OPTS}
)

# add dependencies
target_add_dependencies(TARGET ${PROJECT_NAME}
    DEPENDS
        mdl::mdl_core
        mdl_core::shared
        cuda
        openimageio
        )

# creates a user settings file to setup the debugger (visual studio only, otherwise this is a no-op)
target_create_vs_user_settings(TARGET ${PROJECT_NAME})

# -------------------------------------------------------------------------------------------------
# PTX Target
# -------------------------------------------------------------------------------------------------
target_add_cuda_ptx_rule(TARGET ${PROJECT_NAME}
    DEPENDS
        mdl::mdl_core
        mdl_core::shared
    CUDA_SOURCES
        "example_execution_cuda.cu"
        "example_execution_cuda_derivatives.cu"
    FAST_MATH
    )

# create installation rules
add_target_install(
    TARGET ${PROJECT_NAME}
    DESTINATION "examples/mdl_core/execution_cuda"
    INCLUDE "*.ptx"
    )
